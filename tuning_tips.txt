;WITH BRGU_Filtered AS
(
    SELECT
        BRGU_GU_ID,
        BRGU_USAGE_ID,
        RLTD_RULE_TYPE
    FROM dbo.BPA_BRGU_RUL_GRP_U
    WHERE RLTD_RULE_TYPE IN (?)
      AND BRGU_USAGE_ID IN (?)
),
VBMP_Filtered AS
(
    SELECT
        RLBP_GU_ID,
        VBMP_VALUE
    FROM dbo.BPA_VBMP_VBP_MAP
    WHERE VBMP_TAG = ?
      AND VBMP_VALUE IN (?)
)
SELECT
    RLBP.RLBP_GU_ID,
    BRGU.BRGU_USAGE_ID AS [PREFIX],
    BRGU.RLTD_RULE_TYPE AS [SPCR OR SRCR],
    SHDR.SHDR_ID AS [BASE ID],
    RLBP.RLBP_SECTION AS [SECTION],
    RLBP.RLBP_SEQ_NO AS [SEQUENCE],
    QLGR.QLGR_GRP_DESC AS [QUALIFIER GRP],
    RLBP.RLBP_DESC AS [RULE DESCRIPTION],
    CAST(RLBP.RLBP_EFF_DT  AS date) AS [EFFECTIVE DATE],
    CAST(RLBP.RLBP_TERM_DT AS date) AS [TERMINATION DATE],
    VBMP.VBMP_VALUE AS [SUPP SEID],
    VBMP1.VBMP_VALUE AS RPTC,
    QLGR.QLGR_AND_OR_IND AS [AND OR INDICATOR],
    RPTC.RPTC_DESC AS [RPTC DESC]
FROM BRGU_Filtered AS BRGU
JOIN dbo.BPA_BRGR_RUL_GRP_R AS BRGR
    ON BRGR.BRGR_GU_ID = BRGU.BRGU_GU_ID
JOIN dbo.BPA_RLGD_RUL_GRP_D AS RLGD
    ON RLGD.RLGD_GU_ID = BRGR.RLGD_GU_ID
   AND RLGD.RLTD_RULE_TYPE = BRGU.RLTD_RULE_TYPE
   AND RLGD.RLGD_BRGU_GRP_ID = BRGU.BRGU_USAGE_ID
JOIN dbo.BPA_RLGR_GRP_SET_R AS RLGR
    ON RLGR.RLGD_GU_ID = RLGD.RLGD_GU_ID
JOIN dbo.BPA_RLBP_BP_RULE_D AS RLBP
    ON RLBP.RLBP_GU_ID = RLGR.RLBP_GU_ID
JOIN dbo.BPA_QLCD_COND_D AS QLCD
    ON QLCD.QLCD_GU_ID = RLBP.QLCD_GU_ID
JOIN dbo.BPA_QLGR_REL_R AS QGRL
    ON QGRL.QLCD_GU_ID = QLCD.QLCD_GU_ID
JOIN dbo.BPA_QLGR_QGRP_D AS QLGR
    ON QLGR.QLGR_GU_ID = QGRL.QLGR_GU_ID
   AND QLGR.RLTD_RULE_TYPE = RLBP.RLTD_RULE_TYPE
   AND QLGR.QLTD_QUAL_TYPE = RLGD.QLTD_QUAL_TYPE
JOIN VBMP_Filtered AS VBMP
    ON VBMP.RLBP_GU_ID = RLBP.RLBP_GU_ID
JOIN dbo.BPA_VBMP_VBP_MAP AS VBMP1
    ON VBMP1.RLBP_GU_ID = RLBP.RLBP_GU_ID
JOIN dbo.CMC_RPTC_DESC AS RPTC
    ON RPTC.RPTC_RPT_CAT = VBMP1.VBMP_VALUE
JOIN dbo.BPA_SHDR_SECTION_HDR AS SHDR
    ON SHDR.RLBP_GU_ID = RLBP.RLBP_GU_ID
   AND SHDR.RLTD_RULE_TYPE = BRGU.RLTD_RULE_TYPE
   AND SHDR.SHDR_GROUP_ID = BRGU.BRGU_USAGE_ID
ORDER BY
    SHDR.SHDR_ID,
    RLBP.RLBP_TERM_DT,
    BRGU.BRGU_USAGE_ID,
    BRGU.RLTD_RULE_TYPE,
    RLBP.RLBP_SECTION,
    RLBP.RLBP_SEQ_NO;

SELECT
    xd.value('(event/@timestamp)[1]',                                  'datetime2(3)')  AS event_time_utc,
    xd.value('(event/@name)[1]',                                       'varchar(60)')   AS event_name,
    @@SERVERNAME                                                                         AS server_instance_name,
    xd.value('(event/action[@name="database_name"]/value)[1]',         'sysname')       AS database_name,
    xd.value('(event/action[@name="server_principal_name"]/value)[1]', 'sysname')       AS server_principal_name,
    xd.value('(event/action[@name="client_app_name"]/value)[1]',       'nvarchar(128)') AS client_app_name,
    xd.value('(event/action[@name="client_hostname"]/value)[1]',       'nvarchar(128)') AS client_hostname,
    xd.value('(event/data[@name="client_ip"]/value)[1]',               'varchar(48)')   AS client_ip_address,
    xd.value('(event/action[@name="session_id"]/value)[1]',            'int')           AS session_id,
    CASE
        WHEN xd.value('(event/@name)[1]', 'varchar(60)') = 'login'
        THEN TRY_CAST(
                xd.value('(event/data[@name="success"]/value)[1]', 'varchar(5)')
             AS bit)
        ELSE NULL
    END                                                                                  AS is_success,
    f.file_name                                                                          AS xe_file_name
FROM sys.fn_xe_file_target_read_file(N'D:\SQLXE\UserActivityAudit*.xel', NULL, NULL, NULL) AS f
CROSS APPLY (SELECT CAST(f.event_data AS xml)) AS x(xd)
ORDER BY
    xd.value('(event/@timestamp)[1]', 'datetime2(3)') DESC;
GO
